#include <stdio.h>
int main() {
    int P, N, R;
    printf("Enter the number of processes (P): ");
    scanf("%d", &P);
    printf("Enter the maximum number of resources each process may need (N): ");
    scanf("%d", &N);
    R = P * (N - 1) + 1;
    printf("Minimum number of resources needed to avoid deadlock: %d\n", R);
    return 0;
}
